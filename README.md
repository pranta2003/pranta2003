<script>
  // Small, self-contained widget that uses GitHub REST API v3.
  // Behavior:
  // 1. Fetch /repos/:owner/:repo for core numbers (stars, forks, watchers, size, updated_at, open_issues_count)
  // 2. Try to estimate commit count by requesting commits?per_page=1 and parsing Link header to find last page number.
  //    If Link header not available, fallback to summing contributors (may require pagination and token).

  const fetchBtn = document.getElementById('fetchBtn');
  const autoBtn = document.getElementById('autoBtn');
  const ownerRepoInput = document.getElementById('ownerRepo');
  const tokenInput = document.getElementById('token');
  const errorBox = document.getElementById('error');
  const statsBox = document.getElementById('stats');
  const updatedEl = document.getElementById('updated');
  let autoInterval = null;

  fetchBtn.addEventListener('click', () => runFetch());
  autoBtn.addEventListener('click', toggleAuto);

  async function runFetch(){
    hideError();
    statsBox.style.display = 'none';
    const or = ownerRepoInput.value.trim();
    if(!or || !or.includes('/')) return showError('Enter repository in owner/repo format.');
    const [owner, repo] = or.split('/').map(s=>s.trim());
    const token = tokenInput.value.trim();
    try{
      setLoading(true);
      const headers = { 'Accept': 'application/vnd.github.v3+json' };
      if(token) headers['Authorization'] = 'token '+token;

      // 1) core repo info
      const repoResp = await fetch(`https://api.github.com/repos/${owner}/${repo}`, { headers });
      if(!repoResp.ok) throw new Error('Repo fetch failed: '+repoResp.status+' '+repoResp.statusText);
      const repoData = await repoResp.json();

      // update basic UI
      document.getElementById('repoName').textContent = repoData.full_name;
      document.getElementById('desc').textContent = repoData.description || '';
      document.getElementById('stars').textContent = numberWithCommas(repoData.stargazers_count||0);
      document.getElementById('forks').textContent = numberWithCommas(repoData.forks_count||0);
      document.getElementById('watchers').textContent = numberWithCommas(repoData.watchers_count||0);
      document.getElementById('issues').textContent = numberWithCommas(repoData.open_issues_count||0);
      document.getElementById('size').textContent = numberWithCommas(repoData.size||0);
      updatedEl.textContent = new Date().toLocaleString();

      // 2) try to get commit count from commits endpoint via Link header trick
      const defaultBranch = repoData.default_branch || 'main';
      const commitsResp = await fetch(`https://api.github.com/repos/${owner}/${repo}/commits?sha=${defaultBranch}&per_page=1`, { headers });
      if(!commitsResp.ok) throw new Error('Commits fetch failed: '+commitsResp.status+' '+commitsResp.statusText);

      const link = commitsResp.headers.get('Link');
      let commitCount = null;
      if(link){
        // parse last page from link header
        const m = link.match(/<([^>]+)>; rel="last"/);
        if(m && m[1]){
          const url = new URL(m[1]);
          commitCount = parseInt(url.searchParams.get('page')) || null;
        }
      }

      if(commitCount === null){
        // fallback: use contributors endpoint and sum contributions (best-effort, may require pagination)
        commitCount = await estimateCommitsFromContributors(owner, repo, headers);
      }

      document.getElementById('commits').textContent = commitCount!==null ? numberWithCommas(commitCount) : 'N/A';
      statsBox.style.display = 'block';
    }catch(err){
      showError(err.message);
    }finally{ setLoading(false); }
  }

  function setLoading(v){
    fetchBtn.disabled = v; autoBtn.disabled = v;
    fetchBtn.textContent = v ? 'Fetching...' : 'Fetch';
  }

  function showError(msg){ errorBox.style.display='block'; errorBox.textContent = msg; }
  function hideError(){ errorBox.style.display='none'; errorBox.textContent = ''; }

  function numberWithCommas(x){ if(x===null||x===undefined) return '—'; return x.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ","); }

  async function estimateCommitsFromContributors(owner, repo, headers){
    try{
      let page = 1; let total = 0; let keep = true;
      while(keep && page < 50){ // safety cap to avoid infinite loop
        const resp = await fetch(`https://api.github.com/repos/${owner}/${repo}/contributors?per_page=100&page=${page}&anon=true`, { headers });
        if(!resp.ok) { keep=false; break; }
        const data = await resp.json();
        if(!Array.isArray(data) || data.length===0){ keep=false; break; }
        for(const c of data){ total += (c.contributions||0); }
        // if less than 100 items, last page
        if(data.length < 100) keep=false; else page++;
      }
      return total>0? total : null;
    }catch(e){ return null; }
  }

  function toggleAuto(){
    if(autoInterval){ clearInterval(autoInterval); autoInterval = null; autoBtn.textContent='Auto: Off'; autoBtn.classList.add('secondary');
    } else {
      // default auto every 60s
      autoInterval = setInterval(()=>{ runFetch(); }, 60000);
      autoBtn.textContent='Auto: On (60s)'; autoBtn.classList.remove('secondary');
      // immediately run once
      runFetch();
    }
  }

  // Allow pressing Enter in owner/repo to fetch
  ownerRepoInput.addEventListener('keydown', (e)=>{ if(e.key==='Enter') runFetch(); });
  tokenInput.addEventListener('keydown', (e)=>{ if(e.key==='Enter') runFetch(); });

</script>
